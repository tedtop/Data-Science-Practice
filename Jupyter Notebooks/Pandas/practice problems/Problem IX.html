september_data = df[df.index.month == 9]

data_2010_2019 = september_data['2010':'2019']

# Plot the density for TMAX
plt.figure(figsize=(12, 6))
data_1950_1959['TMAX'].plot.density(label='1950-1959 TMAX', legend=True)
data_2010_2019['TMAX'].plot.density(label='2010-2019 TMAX', legend=True)

data_2010_2019['TMIN'].plot.density(label='2010-2019 TMIN', legend=True)

plt.ylabel('Density')
plt.show()
